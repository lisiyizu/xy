    # 隐藏证件号中间数字

    隐藏证件号中间数字，只保留首尾两处各三位数字

    # 示例：

    233****234

    # 代码实现：

    ...
<div>{{ID | hideIDMiddle}}</div>
...  

<script>
import { hideIDMiddle } from "@xiyun/utils";

export default {
    filters: { hideIDMiddle },
    data() {
        return {
            ID: '233205199210121234',
        }
    },
}
</script>
